	title = authoritarianism_ideologie_events.3.t
	desc = authoritarianism_ideologie_events.3.d
	picture = GFX_report_event_fascist_gathering

	trigger = {
		has_idea_with_trait = authoritarianism_ideologie_demagogue
		NOT = { has_government = authoritarianism_ideologie }
		has_country_flag = authoritarianism_ideologie_government_support
		authoritarianism_ideologie > 0.3
		is_major = no
		is_puppet = no
	}

	mean_time_to_happen = {
		days = 100
		modifier = {
			factor = 0.5
			authoritarianism_ideologie > 0.4
		}
	}

	immediate = {
		hidden_effect = { remove_ideas = authoritarianism_ideologie_revolutionaries }
	}

	option = {
		name = authoritarianism_ideologie_events.3.a
		set_politics = {
			ruling_party = authoritarianism_ideologie
			elections_allowed = no
		}
		add_timed_idea = { idea = reign_of_terror days = 730 }
	}
}
