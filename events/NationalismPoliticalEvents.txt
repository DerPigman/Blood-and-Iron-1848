# Move Towards Nationalism
country_event = {
  id = nationalism_ideologie_events.1
  title = nationalism_ideologie_events.1.t
  desc = nationalism_ideologie_events.1.d
	picture = GFX_report_event_journalists_speech

	is_triggered_only = yes

	trigger = {
		NOT = {
      OR = {
        	has_government = conservatism_ideologie
	has_government = liberalism_ideologie
	has_government = centrism_ideologie
	has_government = socialism_ideologie
	has_government = nationalism_ideologie
	has_government = oligarchism_ideologie

      }
    }
	}

	# Seek government support
	option = {
		name = nationalism_ideologie_events.1.a
		custom_effect_tooltip = nationalism_ideologie_events.1.a.tt
		clr_country_flag = nationalism_ideologie_popular_movement
		clr_country_flag = nationalism_ideologie_opposition_formed
		set_country_flag = nationalism_ideologie_political_movement
	}

	# Seek popular support
	option = {
		name = nationalism_ideologie_events.1.b
		custom_effect_tooltip = nationalism_ideologie_events.1.b.tt
		clr_country_flag = nationalism_ideologie_political_movement
		clr_country_flag = nationalism_ideologie_government_support
		set_country_flag = nationalism_ideologie_popular_movement
	}
}

country_event = {
	id = nationalism_ideologie_events.5
	title = nationalism_ideologie_events.5.t
	desc = nationalism_ideologie_events.5.d
	picture = GFX_report_event_election_vote

	trigger = {
		has_idea_with_trait = nationalism_ideologie_reformer
    NOT = {
      OR = {
        	has_government = conservatism_ideologie
	has_government = liberalism_ideologie
	has_government = centrism_ideologie
	has_government = socialism_ideologie
	has_government = nationalism_ideologie
	has_government = oligarchism_ideologie

      }
    }
		nationalism_ideologie > 0.5
		has_country_flag = nationalism_ideologie_government_support
		is_puppet = no
	}

	mean_time_to_happen = {
		days = 200
	}

	option = {
		name = nationalism_ideologie_events.5.a
    if = {
			limit = {
        OR = {
          	has_government = authoritarianism_ideologie

        }
      }
			add_timed_idea = { idea = fascism_defeated days = 730 }
		}
		if = {
			limit = {
        OR = {
          	has_government = jacobinism_ideologie
	has_government = marxism_ideologie

        }
      }
			add_timed_idea = { idea = communism_defeated days = 730 }
		}
		set_politics = {
			ruling_party = nationalism_ideologie
			elections_allowed = yes
		}
		hold_election = ROOT
		remove_ideas = nationalism_ideologie_revolutionaries
	}
	option = {
		name = nationalism_ideologie_events.5.b
	}
}
