country_event = {
	id = jacobinism_ideologie_events.3
	title = jacobinism_ideologie_events.3.t
	desc = jacobinism_ideologie_events.3.d
	picture = GFX_report_event_gathering_protest

	trigger = {
		has_idea_with_trait = jacobinism_ideologie_revolutionary
		NOT = { has_government = jacobinism_ideologie }
		has_country_flag = jacobinism_ideologie_government_support
		jacobinism_ideologie > 0.4
		is_major = no
		is_puppet = no
	}

	mean_time_to_happen = {
		days = 100
		modifier = {
			factor = 0.5
			jacobinism_ideologie > 0.5
		}
	}

	immediate = {
		hidden_effect = { remove_ideas = jacobinism_ideologie_revolutionaries }
	}

	option = {
		name = jacobinism_ideologie_events.3.a
		set_politics = {
			ruling_party = jacobinism_ideologie
			elections_allowed = no
		}
		add_timed_idea = { idea = reign_of_terror days = 730 }
	}
}
