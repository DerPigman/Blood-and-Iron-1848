# Move Towards Conservatism
country_event = {
  id = political.BaI.1
  title = political.BaI.1.t
  desc = political.BaI.1.d
	picture = GFX_report_event_journalists_speech

	is_triggered_only = yes

	trigger = {
  NOT = { has_idea_with_trait = ignore_political_events } NOT = { has_idea_with_trait = ignore_political_events_democratic }
		NOT = {
      OR = {
        	has_government = conservatism_ideology
	has_government = liberalism_ideology
	has_government = centrism_ideology
	has_government = socialism_ideology
	has_government = nationalism_ideology
	has_government = oligarchism_ideology

      }
    }
	}

	# Seek government support
	option = {
		name = political.BaI.1.a
		custom_effect_tooltip = political.BaI.1.a.tt
		clr_country_flag = conservatism_ideology_popular_movement
		clr_country_flag = conservatism_ideology_opposition_formed
		set_country_flag = conservatism_ideology_political_movement
	}

	# Seek popular support
	option = {
		name = political.BaI.1.b
		custom_effect_tooltip = political.BaI.1.b.tt
		clr_country_flag = conservatism_ideology_political_movement
		clr_country_flag = conservatism_ideology_government_support
		set_country_flag = conservatism_ideology_popular_movement
	}
}

	}
}

		}
		hold_election = ROOT

	}
	option = {
		name = political.BaI.5.b
	}
